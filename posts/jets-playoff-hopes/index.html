Looking to break the longest active playoff drought, the New York Jets have found themselves in the tightest division in football with a chance to sneak in to a wild card spot but might need a little help.

library(tidyverse)
library(gt)

espn <- read_csv("~/Documents/SPMC 350/blog/posts/jets-playoff-hopes/espn_data/standings.csv")

standings <- espn %>%
  group_by(Tm) %>%
  select(Tm, W, L, Position, Reason) %>%
  arrange(desc(W)) %>%
  ungroup()

standings %>% 
  gt() %>%
  cols_label(
    Tm = "Team"
  ) %>%
  sub_missing(
    columns = everything(), 
    missing_text = "") %>%
  tab_header(
    title = "Current standings in the AFC (Week 12)",
    subtitle = "The Jets have control over the final Wild Card spot after a Week 12 win"
  ) %>%
  tab_style(
    style = cell_text(color = "black", weight = "bold", align = "center"),
    locations = cells_title("title")
  ) %>% tab_style(
    style = cell_text(color = "black", align = "center"),
    locations = cells_title("subtitle")
  ) %>%
  tab_source_note(
    source_note = md("By: Jack Coil |  Source: Pro Football Reference")
  ) %>%
  tab_style(
     locations = cells_column_labels(columns = everything()),
     style = list(
       cell_borders(sides = "bottom", weight = px(3)),
       cell_text(weight = "bold", size=12)
     )
   ) %>%
  opt_row_striping() %>% 
  opt_table_lines("none") %>%
  tab_style(
    style = list(
      cell_fill(color = "black"),
      cell_text(color = "white")
      ),
    locations = cells_body(
      rows = Tm == "New York Jets")
  )

library(tidyverse)
library(dplyr)
library(ggrepel)
library(ggplot2)

power <- read_csv("~/Documents/SPMC 350/blog/posts/jets-playoff-hopes/espn_data/rating.csv")

jets <- power %>%
  filter(Team == "NYJ")

bills <- power %>%
  filter(Team == "BUF")

ben <- power %>%
  filter(Team == "CIN")

mia <- power %>%
  filter(Team == "MIA")

kc <- power %>%
  filter(Team == "KC")

bal <- power%>%
  filter(Team == "BAL")

ten <- power %>%
  filter(Team == "TEN")

ggplot() + 
  geom_point(data=power, aes(x=FPI, y=`REM SOS`)) + 
  geom_point(
    data=jets, label = "NYJ",
    aes(x=FPI, y=`REM SOS`, size=""), 
    color="red") +
  geom_point(
    data=bills, label = "BUF",
    aes(x=FPI, y=`REM SOS`, size=""), 
    color="red") +
  geom_point(
    data=ben, label = "CIN",
    aes(x=FPI, y=`REM SOS`, size=""), 
    color="red") +
  geom_point(
    data=mia, label = "MIA",
    aes(x=FPI, y=`REM SOS`, size=""), 
    color="blue") +
  geom_point(
    data=kc, label = "KC",
    aes(x=FPI, y=`REM SOS`, size=""), 
    color="blue") +
  geom_point(
    data=bal, label = "BAL",
    aes(x=FPI, y=`REM SOS`, size=""), 
    color="blue") +
  geom_point(
    data=ten, label = "TEN",
    aes(x=FPI, y=`REM SOS`, size=""), 
    color="blue") +
  geom_text(aes(x=9, y=6, label="Good FPI\nWeak SOS"), color="blue") +
  geom_text(aes(x=-9, y=30, label="Bad FPI\nStrong SOS"), color="blue") +
  geom_text_repel(
    data=power, 
    aes(x=FPI, y=`REM SOS`, label=Team)) +
labs(
    x = "Football Power Index (FPI)",
    y = "SOS Remaining",
    caption = "Data: ESPN | By: Jack Coil",
    title = "Using the Power Index and REM SOS to show how teams might fare in the final weeks",
    subtitle = "The Jets at 7-4 have the lowest FPI of all playoff teams but an easier remaining SOS than most of the AFC.",
    label = "New York Jets"
  ) +
  theme_bw() +
  theme(
    plot.title = element_text(size = 12, hjust = 0.5, face = "bold"),
    plot.subtitle = element_text(size = 8, hjust = 0.5)
